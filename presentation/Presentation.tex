        \begin{alertblock}{Physical issue}
        \scriptsize
Superconducting qubits are \textbf{weakly anharmonic}, so noise or fast control can easily excite unwanted higher levels. This phenomenon is called \textbf{leakage}. 
\end{alertblock}
        \begin{block}{Analog quantum control}
        \scriptsize
        Implementation of a quantum operation by injecting control pulses at the Hamiltonian level.
        % , rather than decomposing it into a sequence of discrete logic gates.
        \end{block}

\begin{frame}{Physical model: gmon Hamiltonian and control channels}
\small
The effective Hamiltonian is
\vspace{-1ex}
{\scriptsize
\[
    \hat{H}(t) =
    \underbrace{\frac{\eta}{2}\sum_{j=1}^{2}\hat{n}_{j}(\hat{n}_{j}-1)}_{\text{anharmonicity}}
    + \underbrace{\ctrl{g(t)}(\hat{a}_{2}^{\dagger}\hat{a}_{1}+\hat{a}_{1}^{\dagger}\hat{a}_{2})}_{\text{tunable coupling}}
    + \underbrace{\sum_{j=1}^{2}\ctrl{\delta_{j}(t)}\hat{n}_{j}}_{\text{local detuning}}
    + \underbrace{\sum_{j=1}^{2} i \ctrl{f_{j}(t)}
    \left(
    \hat{a}_{j}e^{i\ctrl{\phi_{j}(t)}}-\hat{a}_{j}^{\dagger}e^{-i\ctrl{\phi_{j}(t)}}
    \right)}_{\text{microwave drive}}
\]}

\begin{frame}{Why this is a multi-objective control problem}
\small
\begin{alertblock}{Cost Function and Multi-objective nature}
    The cost function is the core metric that the agent is trained to minimize.
    Therefore, the multi-objective nature of the control problem must be explicitly encoded in the cost function.
\end{alertblock}


\begin{block}{Universal Function Optimizaion (UFO) cost function}
    \[
C=
\underbrace{\chi\!\left[1-F(U(T))\right]}_{\text{fidelity penalty}}
+\underbrace{\beta\,L_{\mathrm{tot}}(T)}_{\text{leakage penalty}}
+\underbrace{\mu\,P_{\mathrm{bdry}}}_{\text{boundary penalty}}
+\underbrace{\kappa\,T}_{\text{time penalty}}
\]
\end{block}

\vspace{0.2em}


% \begin{block}{Why multi-objective?}
\begin{itemize}\itemsep0.2em
    % \item A good pulse must be \textbf{accurate}, but also \textbf{fast}, \textbf{regular at the boundaries}, and \textbf{low-leakage}.
    \item These goals are not automatically compatible: reducing one penalty can worsen another.
\end{itemize}
% \end{block}

\begin{frame}
    \frametitle{UFO terms insights}
\begin{alertblock}{Universal Function Optimizaion (UFO) cost function}
    \scriptsize
\[
C=
\underbrace{\chi\!\left[1-F(U(T))\right]}_{\text{fidelity penalty}}
+\underbrace{\beta\,L_{\mathrm{tot}}(T)}_{\text{leakage penalty}}
+\underbrace{\mu\,P_{\mathrm{bdry}}}_{\text{boundary penalty}}
+\underbrace{\kappa\,T}_{\text{time penalty}}
\]
\end{alertblock}

\vspace{0.2em}

\scriptsize
\begin{block}{Fidelity}
% Let $U_{\mathrm{target}} \in U(d_{\mathrm{comp}})$ denote the desired logical gate acting on a $d_{\mathrm{comp}}$-dimensional computational subspace, and 
Being $U(T)$ the unitary evolution generated by a control protocol at final time $T$, and $U_{\mathrm{target}}$ the desired logical gate.
% In the idealized closed-system setting, where the action of the control sequence remains unitary on the relevant subspace, a standard overlap-based figure of merit is the \emph{gate fidelity}
\[F(U(T))=\frac{1}{d_{\mathrm{comp}}^2}\left|\mathrm{Tr}\!\left(U^\dagger(T)\,U_{\mathrm{target}}\right)\right|^2, \qquad \mathrm{with} \quad d_{\mathrm{comp}}=4 \]
\end{block}

\begin{block}{Boundary Penalty}
    \[
    P_{\mathrm{bdry}}=\sum_{t\in\{0,T\}}\!\left[g(t)^2+f_1(t)^2+f_2(t)^2\right]
    \]
    where \(g(t), f_1(t), f_2(t)\) are the amplitude controllers at a given time \(t\). 
    This term forces (by penalizing) the control amplitude to be zero at the pulse boundaries, which can cause unwanted excitations and leakage after the termination of the episode.
\end{block}


\end{frame}


\begin{frame}
    \frametitle{UFO terms insights}
    \begin{block}{Leakage term: physics-informed proxy}
        \scriptsize
        % \small
        Considering the separated Hamiltonian in block form, \(\hat{H}_{\mathrm{od}}(t)\) is the off-diagonal block that couples the computational subspace to the leakage subspace, while \(\Delta(t)\) is the instantaneous energy gap between these subspaces. 
        \[
            L_{\mathrm{tot}}(T)\lesssim
            \frac{\|\hat{H}_{\mathrm{od}}(0)\|}{\Delta(0)}
            +
            \frac{\|\hat{H}_{\mathrm{od}}(T)\|}{\Delta(T)}
            +
            \int_{0}^{T}
\frac{1}{\Delta(t)^2}
\left\|
\frac{d^2 \hat{H}_{\mathrm{od}}(t)}{dt^2}
\right\|dt
\]
\scriptsize


The first two terms penalize residual couplings at the pulse boundaries; the integral term penalizes rapid temporal variations that can induce non-adiabatic transitions, effectively working as a smoothing filter.

\end{block}
\begin{block}{Time penalty}
    \scriptsize
    Since the experiment values do not coincide with the natural time units of the system, a normalization factor is introduced.
    \[
    T = \frac{t}{T_{\mathrm{norm}}}, \quad \text{with } T_{\mathrm{norm}}=60\,\mathrm{ns}
    \]
\end{block}
\end{frame}
