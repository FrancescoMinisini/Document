\begin{tikzpicture}
    % Define common options
    \pgfplotsset{

            scale only axis,
            height=4cm,
            grid=major,
            grid style={dashed, gray!30},
            xmin=-0.1, xmax=3.25,
            xtick={0, 1.0, 2.0, 3.0},

            legend style={font=\footnotesize, inner sep=2pt, rounded corners=2pt},
            xlabel={$\alpha$},
        }

    \begin{axis}[
        name=plot_qual,
        at={(plot_hor.south east)},

        xshift=2cm,
        width=0.35\linewidth,
        ylabel=\textcolor{red!80!black}{Nominal Fidelity},
        title={(b) Nominal Quality},

        ymin=0.935, ymax=1.005,
        yticklabel style={font=\footnotesize, color=red!80!black},
    ]
    \addplot[
        color=red!80!black, mark=*, thick, mark size=1.5pt
    ] table[x=alpha, y=best_fidelity, col sep=comma] {data/adam/adam_family_sweep.csv};
    \end{axis}
    
    \begin{axis}[
        at={(plot_hor.south east)},
        anchor=south west,

    ] table[x=alpha, y=best_cost, col sep=comma] {data/adam/adam_family_sweep.csv};

    \begin{axis}[
        name=plot_rob,
        at={($(plot_hor.south east)!0.5!(plot_qual.south west)$)},

        yshift=-1.8cm,
        width=0.65\linewidth,

        title={(c) Robustness},
        axis y line*=left,

        yticklabel style={font=\footnotesize, color=black!70!green},

    \addplot[
        color=black!70!green, thick,
        mark=*, mark options={fill=black!70!green}, mark size=1.5pt,
    ] table[x=alpha, y=robustness_sigma_1p0_average_fidelity, col sep=comma] {data/adam/adam_family_sweep.csv};

    \begin{axis}[

        anchor=north,
        yshift=-1.8cm,

        axis y line*=right,
        axis x line=none,
        yticklabel={\pgfmathprintnumber[fixed,fixed zerofill,precision=4]{\tick}},
        scaled y ticks=false,
        yticklabel style={font=\footnotesize, color=blue!60!magenta},

    \addplot[
        color=blue!60!magenta, mark=x, only marks, mark size=2pt
    ] table[x=alpha, y=robustness_sigma_1p0_fidelity_variance, col sep=comma] {data/adam/adam_family_sweep.csv};
